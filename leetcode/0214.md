class Solution(object):
    def shortestPalindrome(self, s):
        """
        :type s: str
        :rtype: str
        """
        def gen_pmt(s):
            n = len(s)
            pmt = [0] * n
            i, j = 1, 0
            while i < n:
                if s[i] == s[j]:
                    pmt[i] = j + 1
                    j += 1
                    i += 1
                elif j == 0:
                    i += 1
                else:
                    j = pmt[j - 1]
            return pmt
        
        if not s: return ""
        pmt = gen_pmt(s + "#" + s[::-1])
        return s[pmt[-1]:][::-1] + s
